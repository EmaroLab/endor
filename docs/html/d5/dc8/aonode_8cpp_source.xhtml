68 void AOnode::isFeasible()
69 {
70  bool temp_isFeasible = false;
71 
72  // 1. the node is feasible if it is already feasible
73  if (nFeasible == true)
74  temp_isFeasible = true;
75 
76  // 2. the node is feasible if it is a terminal node (no hyperarcs)
77  if (arcs.size() == 0)
78  temp_isFeasible = true;
79 
80  // 3. the node is feasible if it has >=1 hyperarcs with all child nodes solved
81  // iterate on the hyperarcs of the node
82  for (int i=0; i<(int)arcs.size(); i++)
83  {
84  bool allSolved = true;
85 
86  // iterate on the child nodes
87  for (int j=0; j<(int)arcs[i].children.size(); j++)
88  {
89  // if the node is not solved --> break
90  if (arcs[i].children[j]->nSolved == false)
91  {
92  allSolved = false;
93  break;
94  }
95  }
96  // if all the child nodes are solved, this node is feasible
97  if (allSolved == true)
98  {
99  temp_isFeasible = true;
100  break;
101  }
102  }
103 
104  // set the feasibility property
105  nFeasible = temp_isFeasible;
106 }
