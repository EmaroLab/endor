79 void AOnode::isFeasible()
80 {
81  bool temp_isFeasible = false;
82 
83  // 1. the node is feasible if it is already feasible
84  if (nFeasible == true)
85  temp_isFeasible = true;
86 
87  // 2. the node is feasible if it is a terminal node (no hyperarcs)
88  if (arcs.size() == 0)
89  temp_isFeasible = true;
90 
91  // 3. the node is feasible if it has >=1 hyperarcs with all child nodes solved
92  // iterate on the hyperarcs of the node
93  for (int i=0; i<(int)arcs.size(); i++)
94  {
95  bool allSolved = true;
96 
97  // iterate on the child nodes
98  for (int j=0; j<(int)arcs[i].children.size(); j++)
99  {
100  // if the node is not solved --> break
101  if (arcs[i].children[j]->nSolved == false)
102  {
103  allSolved = false;
104  break;
105  }
106  }
107  // if all the child nodes are solved, this node is feasible
108  if (allSolved == true)
109  {
110  temp_isFeasible = true;
111  break;
112  }
113  }
114 
115  // set the feasibility property
116  nFeasible = temp_isFeasible;
117 }
